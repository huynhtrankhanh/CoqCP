Fixpoint multiply (n : nat) (l : list nat) :=
  match n with
  | 0 => []
  | S n => l ++ multiply n l
  end.

Fixpoint lexLess (l1 l2 : list nat) :=
  match l1, l2 with
  | _, [] => false
  | [], (_ :: _) => true
  | (a :: b), (a' :: b') => bool_decide (a < a') || (bool_decide (a = a') && lexLess b b')
  end.
